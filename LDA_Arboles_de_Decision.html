import pandas as pd
import numpy as np
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LassoCV

df = pd.read_csv("teen_phone_addiction_dataset.csv")
df = df.drop(columns=["Name","Location","School_Grade"], errors="ignore")
df_encoded = pd.get_dummies(df, drop_first=True)

Q1 = df_encoded["Addiction_Level"].quantile(0.25)
Q3 = df_encoded["Addiction_Level"].quantile(0.75)
IQR = Q3 - Q1
mask = (df_encoded["Addiction_Level"] < Q1 - 1.5*IQR) | (df_encoded["Addiction_Level"] > Q3 + 1.5*IQR)
df_no_outliers = df_encoded.loc[~mask].copy()

df_imputed = df_no_outliers.fillna(df_no_outliers.mean(numeric_only=True))

X_tmp = df_imputed.drop(columns=["Addiction_Level","ID"], errors="ignore")
y_cont = df_imputed["Addiction_Level"]
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_tmp)
lasso = LassoCV(cv=5, random_state=42).fit(X_scaled, y_cont)
selected = X_tmp.columns[(lasso.coef_ != 0)].tolist()

threshold = df_imputed["Addiction_Level"].median()
y_binary = np.where(df_imputed["Addiction_Level"] > threshold, 1, 0)

y_binary = pd.qcut(
    df_imputed["Addiction_Level"].rank(method="first"),
    q=[0, 0.5, 1.0],
    labels=[0, 1]
).astype(int)

X = df_imputed[selected].copy()
y = pd.Series(y_binary, name="target")

def proporciones(yvec, titulo):
    p = yvec.value_counts(normalize=True).sort_index().reindex([0,1], fill_value=0.0)
    print(f"{titulo} -> 0:{p.loc[0]:.3f} | 1:{p.loc[1]:.3f} (n={len(yvec)})")

proporciones(y, "Datos originales")

from sklearn.model_selection import StratifiedShuffleSplit

def split_con_dos_clases(X, y, test_size=0.20, intentos=30, seed=42):
    sss = StratifiedShuffleSplit(n_splits=intentos, test_size=test_size, random_state=seed)
    for tr_idx, te_idx in sss.split(X, y):
        y_tr, y_te = y.iloc[tr_idx], y.iloc[te_idx]
        if y_tr.nunique() == 2 and y_te.nunique() == 2:
            return X.iloc[tr_idx], X.iloc[te_idx], y_tr, y_te
    return None, None, None, None

X_train, X_test, y_train, y_test = split_con_dos_clases(X, y, test_size=0.20)

if X_train is None:
    X_train, X_test, y_train, y_test = split_con_dos_clases(X, y, test_size=0.10)

if X_train is None:
    raise ValueError("No fue posible crear train/test con ambas clases. Revisa la binarizacin o el tamao de la clase minoritaria.")

proporciones(y_train, "Entrenamiento")
proporciones(y_test,  "Prueba")

import time
from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.model_selection import KFold, cross_val_score

# Definir clasificador base
clf = DecisionTreeClassifier(random_state=42)

# Obtener valores de complejidad alfa candidatos
path = clf.cost_complexity_pruning_path(X_train_sel, y_train)
ccp_alphas = path.ccp_alphas

# K-Fold CV para seleccionar mejor alpha
kf = KFold(n_splits=10, shuffle=True, random_state=42)
mejor_alpha = None
mejor_score = -np.inf

t0 = time.time()
for alpha in ccp_alphas:
    clf_tmp = DecisionTreeClassifier(random_state=42, ccp_alpha=alpha)
    scores = cross_val_score(clf_tmp, X_train_sel, y_train, cv=kf)
    score_mean = scores.mean()
    if score_mean > mejor_score:
        mejor_score = score_mean
        mejor_alpha = alpha
t1 = time.time()

print(f"Mejor alpha: {mejor_alpha:.6f} con score={mejor_score:.3f}")
print(f"Tiempo K-Fold CV: {t1 - t0:.2f} segundos")

# Entrenar rbol definitivo con mejor alpha
clf_final = DecisionTreeClassifier(random_state=42, ccp_alpha=mejor_alpha)
clf_final.fit(X_train_sel, y_train)

# Visualizar rbol
plt.figure(figsize=(12,6))
plot_tree(clf_final, feature_names=X_train_sel.columns, class_names=["0","1"], filled=True)
plt.title("rbol de decisin podado")
plt.show()

# Visualizar particin en el espacio de variables
x_min, x_max = X_train_sel.iloc[:,0].min()-1, X_train_sel.iloc[:,0].max()+1

Conclusin General

A lo largo de los incisos se desarroll un flujo completo de anlisis y modelado. En el primer paso se garantiz una correcta divisin de los datos en conjuntos de entrenamiento y prueba, preservando el balance de clases. Posteriormente, se ajust un modelo de regresin logstica con GLM para identificar las variables ms relevantes, seleccionndose nicamente dos (Daily_usage_hours y Time_on_social_media) como principales predictores.

Con estas variables se implementaron dos metodologas distintas: Linear Discriminant Analysis (LDA) y rboles de Decisin con poda. En ambos casos se generaron visualizaciones que permitieron observar cmo se distribuyen las clases en el espacio de las variables y cmo cada modelo establece fronteras de decisin. Para el rbol, se seleccion un valor ptimo de utilizando validacin cruzada K-Fold, lo que permiti obtener un modelo ms generalizable.

En la fase de evaluacin, se compararon las mtricas de desempeo en el conjunto de prueba. Los resultados mostraron valores muy similares entre ambos modelos, con un accuracy cercano a 0.78 en los dos casos. El LDA present una ligera ventaja en accuracy y recall, lo que indica una mejor capacidad para recuperar instancias de ambas clases. Por su parte, el rbol de decisin alcanz una precisin marginalmente superior, aunque con un pequeo sacrificio en recall.

En conclusin, los dos enfoques son adecuados para esta tarea, pero el LDA puede considerarse ms robusto al ofrecer un equilibrio ms uniforme entre las mtricas, mientras que el rbol de decisin resulta ms interpretable gracias a la estructura jerrquica y las reglas de decisin explcitas. La eleccin entre uno u otro depender de la prioridad: maximizar la recuperacin de instancias correctas (LDA) o contar con un modelo fcilmente interpretable (rbol podado).
